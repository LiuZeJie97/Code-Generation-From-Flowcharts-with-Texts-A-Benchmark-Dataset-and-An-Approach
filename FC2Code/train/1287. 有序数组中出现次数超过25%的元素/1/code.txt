def findSpecialInteger(self, arr: List[int]) -> int:
    n = len(arr)
    cur, cnt = arr[0], 0
    f = False
    for i in range(n):
        if arr[i] == cur:
            cnt += 1
            if cnt * 4 > n:
                f = True
                break
        else:
            cur, cnt = arr[i], 1
    if f:
        return cur
    else:
        return -1
