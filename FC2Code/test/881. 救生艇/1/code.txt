def numRescueBoats(self, people: List[int], limit: int) -> int:
    number = 0
    people.sort()
    light, heavy = 0, len(people) - 1
    while light <= heavy:
        if people[light] + people[heavy] > limit:
            heavy -= 1
        else:
            light += 1
            heavy -= 1
        number += 1
    return number
