def find132pattern(self, array):
    n为数组array的长度
    定义单调栈candidate_k，将数组中最后一项添加进单调栈
    定义变量max_k等于负的float类型的inf
    i为数组下标，初值为数组array的长度减2
    while(当数组下标大于-1)
        if (数组中下标为i的元素小于max_k)
            output:  存在132模式的子序列
        while(单调栈candidate_k非空且数组下标为i的元素大于单调栈中最后一项元素)
            令max_k等于单调栈中最后一项元素
            移除单调栈栈顶元素
        if (数组下标为i的元素大于max_k)
            将该元素添加进单调栈
        else:
        数组下标减一
    output:  不存在132模式的子序列
