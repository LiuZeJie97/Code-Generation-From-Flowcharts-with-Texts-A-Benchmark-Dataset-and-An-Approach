def numSubmat(self, mat):
    n为矩阵mat的行数，m为矩阵mat的列数
    定义n行m列初值均为0的二维数组row
    数组下标i等于0
    while(i小于矩阵mat的行数)
        数组下标j等于0
        while(j小于矩阵的列数)
            if (j等于0)
                数组row第i行第j列的元素等于矩阵第i行第j列的元素
            else:
                若矩阵第i行第j列的元素等于0，则数组row第i行第j列的元素等于0，否则等于数组row第i行第j-1列的元素加一
            j等于j加一
        i等于i加一
    最终答案ans等于0
    j等于0
    while(j小于矩阵mat的列数)
        初始化数组Q
        计算值total等于0
        i等于0
        while(i小于矩阵的行数)
            高度变量height等于1
            while(数组Q非空且Q[-1][0]大于数组row第i行第j列的元素)
                令计算值减去它们的差值与Q[-1][1]的成绩
                令高度变量加上Q[-1][1]
                将数组最后一项元素移除
            令计算值total加上数组row第i行第j列的元素
            令最终答案加上计算值
            将数组row第i行第j列的元素和高度变量组成的元组加入数组Q
            i等于i加一
        j等于j加一
    output:  返回最终答案
