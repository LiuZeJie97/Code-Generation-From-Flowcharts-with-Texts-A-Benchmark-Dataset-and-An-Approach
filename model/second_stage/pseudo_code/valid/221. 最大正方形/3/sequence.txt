def maximalSquare(self, matrix):
    if (矩阵matrix为空)
        output:  返回0。
    else:
        初始化m、n，用于记录矩阵的行数、列数，
        初始化最大边长res等于0。
        初始化dp=[[0, ,0], ,[0, ,0]]，维度为(m+1)(n+1)，dp[i][j]表示以第i行、第j列处为右下角的最大正方形的边长
        i等于1
        while(i小于矩阵的行数加1)
            j等于1
            while(j小于矩阵的列数加1)
                if (若以矩阵第i行、第j列等于"1")
                    以第i+1行、第j+1列处为右下角的最大正方形的边长等于dp[i-1][j-1]、dp[i-1][j]、dp[i][j-1]中的较小值加一
                    并更新最大边长等于其自身和dp[i][j]中的较大值
                else:
                j自增1
            i自增1
        output:  依据最大边长计算正方形的面积并返回
