def sortedSquares(self, number):
    n为数组number的长度
    定义变量negative用于记录负数与非负数的分界位置，初值为-1
    定义变量i，初值为0
    while(i小于数组number的长度)
        if (!(数组中下标为i的元素小于0))
            break
        负数与非负数的分界位置等于i
        i等于i加一
    定义答案数组ans
    定义左指针i，初值为分界位置，定义右指针j，初值为分界位置加一
    while(i大于等于0或j小于数组number的长度)
        if (左指针小于0)
            计算数组右指针元素的2次方，将计算结果添加进答案数组
            右指针加一
            continue
        else:
            if (右指针等于数组的长度)
                计算数组左指针元素的2次方，将计算结果添加进答案数组
                左指针减一
                continue
            else:
                if (数组左指针元素的2次方小于右指针元素的2次方)
                    计算数组左指针元素的2次方，将计算结果添加进答案数组
                    左指针减一
                    continue
                else:
                    计算数组右指针元素的2次方，将计算结果添加进答案数组
                    右指针加一
                    continue
    output:  返回答案数组
