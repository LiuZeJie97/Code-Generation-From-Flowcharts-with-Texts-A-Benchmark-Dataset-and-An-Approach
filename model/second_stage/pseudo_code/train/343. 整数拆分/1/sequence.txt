def integerBreak(self, n):
    if (n小于等于3)
        output:  返回n减1
    else:
        定义变量quotient等于n整除3，变量remainder等于n除以3的余数
        if (n除以3的余数等于0)
            output:  返回3的quotient次方
        else:
            if (n除以3的余数等于1)
                output:  返回3的quotient-1次方乘以4
            else:
                output:  返回3的quotient次方乘以2
