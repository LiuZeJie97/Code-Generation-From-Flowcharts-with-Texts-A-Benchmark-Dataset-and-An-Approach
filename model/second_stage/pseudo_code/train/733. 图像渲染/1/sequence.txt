def floodFill(self, image, sr, sc, newColor):
    定义变量currColor等于数组image中位于sr行sc列的像素值
    if (currColor等于newColor)
        output:  返回数组image
    else:
        n为数组image的行数，m为数组的列数
        定义队列que，并将位置坐标(sr,sc)加入队列
        位于sr行sc列的图画的像素值等于newColor
        while(队列不为空)
            定义x为队头元素的横坐标，y为队头元素纵坐标，并将队头元素出队
            mx为当前横坐标，my为当前纵坐标，(mx,my)表示当前位置坐标
            while(当前位置坐标(mx,my)位于位置坐标(x,y)的左方向1格，右方向1格，上方向1格，下方向1格时)
                if (当前横坐标小于0或当前横坐标大于等于数组image的列数或当前纵坐标小于0或当前纵坐标大于等于数组的行数)
                else:
                    if (数组中位于当前位置坐标的图画的像素值不等于currColor)
                    else:
                        将当前位置坐标添加进队列
                        数组中位于当前位置坐标的图画的像素值更新为newColor
                更新当前位置坐标
        output:  返回数组image
