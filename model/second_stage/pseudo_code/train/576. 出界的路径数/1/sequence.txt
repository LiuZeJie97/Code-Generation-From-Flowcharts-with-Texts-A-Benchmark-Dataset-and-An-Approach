def findPaths(self, m, n, maxMove, startRow, startColumn):
    MOD等于10的9次方加7
    定义初值为0的变量outCounts，用于记录球移出边界的路径数量
    定义三维分别是maxMove+1，m，n的三维数组dp，其初值均为0，dp[i][j][k]表示球移动i次后位于(j,k)的路径数量
    球移动0次后位于(startRow,startColumn)的路径数量为1
    i为数组下标
    while(i小于maxMove)
        j为数组下标
        while(j小于m)
            k为数组下标
            while(k小于n)
                if (球移动i次后位于(j,k)的路径数量大于0)
                    j1为横坐标，k1为纵坐标，(j1,k1)表示位置
                    while(当(j1,k1)分别为位置(j,k)的左方向1格，右方向1格，上方向1格，下方向1格时)
                        if (j1大于等于0小于m且k1大于等于0小于n)
                            球移动i+1次后位于(j1,k1)的路径数量等于该数量与球移动i次后位于(j,k)的路径数量的和对MOD取余
                            continue
                        else:
                            球移出边界的路径数量等于该数量与与球移动i次后位于(j,k)的路径数量的和对MOD取余
                            continue
                else:
                令k加一
            令j加一
        令i加一
    output:  返回球移出边界的路径数量
