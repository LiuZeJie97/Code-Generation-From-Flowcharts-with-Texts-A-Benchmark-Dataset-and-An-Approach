def search(self, a, t):
    if (数组a为空)
        output:  返回False
    else:
        n为数组a的长度
        if (数组的长度为1)
            output:  若数组第一项元素等于目标整数t，则返回True，否则返回False
        else:
            定义l为左边界，初值为0；定义r为右边界，初值为数组的长度减一
            while(左边界小于等于右边界)
                定义中点m等于左边界l与右边界r的和整除2
                if (数组a中点元素的值等于目标整数)
                    output:  返回True
                if (数组中点元素的值等于左边界元素的值且该值等于右边界元素的值)
                    左边界加一
                    右边界减一
                    continue
                else:
                    if (数组左边界元素的值小于数组中点元素的值)
                        if (目标整数大于等于数组左边界l处的值且该整数小于数组中点m处的值)
                            右边界更新为中点减一
                            continue
                        else:
                            左边界更新为中点加一
                            continue
                    else:
                        if (目标整数大于数组中点m处的值且该整数小于等于数组中最后一项的值)
                            左边界更新为中点加一
                            continue
                        else:
                            右边界更新为中点减一
                            continue
            output:  返回False
