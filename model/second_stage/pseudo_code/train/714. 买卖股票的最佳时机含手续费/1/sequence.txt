def maxProfit(self, prices, fee):
    n为数组prices的长度，其中price[i]为第i天的股票价格
    定义变量buy等于第0天股票价格加上交易股票的手续费用，表示最低买入价格
    定义变量profit等于0，表示利润
    i为天数，初值为1
    while(当前天数小于股票价格数组的长度)
        if (第i天的股票价格与交易股票的手续费用fee之和小于最低买入价格)
            将最低买入价格更新为它们的和
        else:
            if (第i天的股票价格大于最低买入价格)
                令利润加上它们的差值
                将最低买入价格更新为第i天的股票价格
            else:
        将当前天数加一
    output:  返回利润
