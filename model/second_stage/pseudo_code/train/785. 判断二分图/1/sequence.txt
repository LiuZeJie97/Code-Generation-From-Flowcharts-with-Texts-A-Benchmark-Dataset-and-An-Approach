def isBipartite(self, graph):
    n为数组graph的长度
    定义初值为0的变量UNCOLORED，初值为1的变量RED，初值为2的变量GREEN
    定义颜色数组color，其长度等于数组graph的长度，初值均为UNCOLORED
    i为数组下标
    while(数组下标小于n)
        if (颜色数组中当前下标元素的值等于UNCOLORED)
            定义队列q，并将i加入队列
            将颜色数组中下标为i的元素的值更新为RED
            while(队列不为空)
                node为队列队头元素，并让队头元素出队
                定义变量cNei，若颜色数组中下标为node的元素的值等于RED，则cNei等于GREEN，否则cNei等于RED
                neighbor为数组graph[node]中的元素
                while(对数组graph[node]进行遍历)
                    if (颜色数组中当前遍历元素下标对应项的值等于UNCOLORED)
                        将当前遍历元素加入队列
                        颜色数组中当前遍历元素下标对应项的值等于cNei
                    else:
                        if (颜色数组中当前遍历元素下标对应项的值不等于cNei)
                            output:  图不是二分图
                    更新被遍历元素
        else:
        数组下标加一
    output:  图是二分图
