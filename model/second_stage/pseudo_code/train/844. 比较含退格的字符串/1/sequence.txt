def backspaceCompare(self, S, T):
    定义字符串S的指针i等于字符串S的长度减一，字符串T的指针j等于字符串T的长度减一
    定义变量skipS表示字符串S中待删除字符的数量，变量skipT表示字符串T中待删除字符的数量，其初值均为0
    while(字符串S的指针大于等于0或字符串T的指针大于等于0)
        while(字符串S的指针大于等于0)
            if (字符串S指针处的字符为"#")
                字符串S中待删除字符的数量加一
                字符串S的指针减一
                continue
            else:
                if (!(字符串S中待删除字符的数量大于0))
                    break
                令该数量减一
                字符串S的指针减一
                continue
        while(字符串T的指针大于等于0)
            if (字符串T指针处的字符为"#")
                字符串T中待删除字符的数量加一
                字符串T的指针减一
                continue
            else:
                if (!(字符串T中待删除字符的数量大于0))
                    break
                令该数量减一
                字符串T的指针减一
                continue
        if (字符串S的指针和字符串T的指针均大于等于0)
            if (字符串S指针处的字符与字符串T指针处的字符不同)
                output:  字符串S和字符串T不相等
        else:
            if (字符串S的指针大于等于0或字符串T的指针大于等于0)
                output:  字符串S和字符串T不相等
        字符串S的指针减一
        字符串T的指针减一
    output:  字符串S和字符串T相等
