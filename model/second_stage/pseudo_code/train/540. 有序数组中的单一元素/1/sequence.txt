def singleNonDuplicate(self, nums):
    定义左边界low等于0，右边界high等于数组nums的长度减一
    while(左边界小于右边界时)
        定义中点pivot等于左边界与右边界的和整除2
        计算中点除以2的余数，令中点减去该余数
        if (若数组nums中点位置元素的值等于中点加1位置元素的值)
            左边界等于中点加2
            continue
        else:
            右边界等于中点
            continue
    output:  返回数组nums左边界位置的元素
