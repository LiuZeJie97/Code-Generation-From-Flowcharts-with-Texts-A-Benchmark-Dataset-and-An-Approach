def minAddToMakeValid(self, s):
    定义变量ans用于记录添加数量，bal用于记录平衡度，其初值均为0
    定义数组下标i，初值为0
    while(数组下标小于字符串s的长度)
        if (如果字符串s中下标为i的字符为'(')
            平衡度加一
        else:
            平衡度减一
        if (平衡度等于-1)
            添加数量加一
            平衡度加一
        else:
        数组下标增加一个单位长度
    output:  返回添加数量与平衡度的和
