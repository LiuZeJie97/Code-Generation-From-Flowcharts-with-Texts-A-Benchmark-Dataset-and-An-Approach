def shipWithinDays(self, weights, days):
    左边界left等于重量数组weights中的最大值，右边界right重量数组中全部元素的和，其中weights[i]表示第i个包裹的重量
    while(左边界小于右边界)
        中值mid等于左边界与右边界的和整除2
        need为需要运送的天数，初值为1
        cur为已经运送的包裹重量之和，初值为0
        i为数组下标，初值为0
        while(数组下标小于重量数组weights的长度)
            if (已经运送的包裹重量之和cur加上第i个包裹的重量大于中值)
                令已经运送的包裹重量之和等于0
                令需要运送的天数need加一
            else:
            令已经运送的包裹重量之和加上第i个包裹的重量
            数组下标自增一个单位
        if (需要运送的天数小于等于目标天数days)
            右边界等于中值mid
            continue
        else:
            左边界等于中值加一
            continue
    output:  返回左边界
