def videoStitching(self, clips, time):
    定义长度为time的数组maxn，数组元素初值均为0
    定义变量last、ret、pre，其初值均为0
    定义变量i
    while(该变量小于二维数组clips的行数)
        定义变量a等于数组clips第i行第0列的元素
        定义变量b等于数组clips第i行第1列的元素
        if (a小于time)
            数组maxn下标为a的元素等于它与b之间的最大值
        else:
        该变量加一
    定义标志变量flag等于False
    定义变量i
    while(该变量小于time)
        last等于它与数组maxn中下标为i的元素之间的最大值
        if (i等于last)
            标志变量flag等于False
            break
        if (i等于pre)
            ret加一
            pre等于last
        else:
        该变量加一
    if (标志变量为True)
        output:  返回ret
    else:
        output:  返回-1
