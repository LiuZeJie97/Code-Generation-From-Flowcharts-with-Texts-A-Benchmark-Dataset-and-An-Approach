def findMin(self, nums):
    定义变量low表示左边界，其初值为0；定义变量high表示右边界，其初值为数组nums的长度减一
    while(左边界小于右边界)
        定义中点pivot等于右边界与左边界的差整除2再加上左边界
        if (数组nums中点元素小于右边界元素)
            右边界更新为中点pivot
            continue
        else:
            if (中点元素大于右边界元素)
                左边界更新为中点加一
                continue
            else:
                右边界减一
                continue
    output:  返回数组左边界元素
