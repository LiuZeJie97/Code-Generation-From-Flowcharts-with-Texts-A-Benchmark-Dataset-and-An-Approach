def numDecodings(self, s):
    n为字符串s的长度
    令a，b，c表示字符串的前i-2，i-1和i个字符的解码方法数，初值为0，1，0
    i等于1，为字符串下标
    while(为字符串下标小于字符串的长度加1)
        前i个字符的解码方法数等于0
        if (字符串下标为i-1的字符不等于"0")
            前i个字符的解码方法数等于其自身加上前i-1个字符的解码方法数
        else:
        if (当前下标大于1，且s[i-2]不等于"0"，且s[i-2:i]小于或等于数字26)
            前i个字符的解码方法数等于其自身加上前i-2个字符的解码方法数
        else:
        令a，b等于b，c
        下标右移
    output:  返回前i个字符的解码方法数
