def removeComments(self, source):
    定义变量in_block等于False
    定义答案数组ans
    i为数组下标
    while(数组下标小于源码数组source的长度)
        定义变量j等于0
        if (变量in_block为False)
            定义数组newline
            while(j小于第i行源码的长度)
                if (第i行源码从位置j到位置j+2的子串内容为'/*'且变量in_block为False)
                    变量in_block等于True
                    j加一
                else:
                    if (上述子串内容为'*/'且变量in_block为True)
                        变量in_block等于False
                        j加一
                    else:
                        if (变量in_block为False且上述子串内容为'//')
                            break
                        if (变量in_block为False)
                            将第i行源码中下标为j的字符添加newline
                        else:
                j加一
            if (newline不为空且变量in_block为False)
                将newline转化为字符串并添加进答案数组中
            else:
            将数组下标右移1个单位
            continue
        else:
    output:  返回答案数组
